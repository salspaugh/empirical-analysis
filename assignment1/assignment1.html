def read_keystrokes(keystroke_filename):
    keystrokes = []
    with open(keystroke_filename) as keystroke_file:
        for line in keystroke_file.readlines():
            record = line.strip().split()
            timestamp = float(record[0])
            user = record[1]
            keystrokes.append((timestamp, user))
    return keystrokes

def plot_interarrival_empirical_pdf(interarrivals, ymax=1.):
    hist, bins = np.histogram(interarrivals, bins=100)
    hist = hist / float(sum(hist))
    plt.plot(bins[1:], hist, color='black')
    plt.ylim(ymin=0, ymax=ymax)
    plt.xlabel("Interarrival Time", fontsize=16)
    plt.ylabel("Empirical Probability", fontsize=16)

    lmbda = 1/np.mean(interarrivals)
    print "Lambda: %f" % lmbda
    dom = range(int(np.ceil(max(interarrivals))))
    exp_pdf = [lmbda * np.exp(-lmbda*x) for x in dom]

def is_exponential_based_on_grubbs(interarrivals):
    Ymin = min(interarrivals)
    Ybar = np.mean(interarrivals)
    s = np.std(interarrivals, ddof=1) # Use unbiased sample standard deviation -- is this right?
    grubbs = (Ybar - Ymin) / s
    print "Grubbs's test statistic: %.5f" % grubbs
    alpha = .05 # significance
    N = len(interarrivals)
    upper_critical_value = scipy.stats.t.pdf(alpha/N, N-2)
    test = ((N-1)/np.sqrt(N)) * np.sqrt(upper_critical_value**2 / (N - 2 + upper_critical_value**2))
    print "Test %.5f:" % test
    is_exponential = grubbs > test # It doesn't follow a normal distribution, it follows an exponential one.
    print "Follows an exponential distribution: %s" % str(is_exponential)
    return is_exponential

_ = is_exponential_based_on_grubbs(user_interarrivals)

def print_pct_in_interval(seq, low=None, high=None):
    n = len(seq)
    if low is None and high is None:
        return 100. 
    if low is not None:
        high_enough = seq > float(low)
        condition = high_enough
    if high is not None:
        low_enough = seq <= float(high)
        condition = low_enough
    if low is not None and high is not None:
        condition = np.logical_and(low_enough, high_enough)
    interval = np.extract(condition, seq)
    pct = len(interval)*1./n*100
    low = "0" if low is None else str(low)
    high = "0" if high is None else str(high)
    print "Percent in interval (%4s, %4s]: \t%.2f" % (low, high, pct)
